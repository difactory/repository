INSERT { 
	GRAPH ?graph { 
	
		?initialState a owl:NamedIndividual , fsm:Initial .
		
		?initialTransition a owl:NamedIndividual , fsm:Transition .
		?initialTransition fsm:Source ?initialState .
		?initialTransition fsm:Target ?state .		
		
		?statemach fsm:contains ?initialState .
		?statemach fsm:contains ?initialTransition .
	}
}
WHERE{

	?statemach rdf:type fsm:StateMachine .
	
	BIND(URI(CONCAT(STR(?state),"_initial") ) AS ?initialState ) . 
	BIND(URI(CONCAT(STR(?state),"_initialTr") ) AS ?initialTransition ) . 

}
