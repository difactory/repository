PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> 
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX list: <https://w3id.org/list#> 
PREFIX express: <https://w3id.org/express#> 
PREFIX ifc: <http://ifcowl.openbimstandards.org/IFC4_ADD1#> 
PREFIX factory: <http://www.ontoeng.com/factory#>
PREFIX osph: <http://www.ontoeng.com/osph#>
PREFIX fsm: <http://www.learninglab.de/~dolog/fsm/fsm.owl#>
PREFIX ifcext: <http://www.ontoeng.com/IFC4_ADD1_extension#>
PREFIX ssn: <http://www.w3.org/ns/ssn/>
PREFIX sosa: <http://www.w3.org/ns/sosa/>
PREFIX stat: <http://www.ontoeng.com/statistics#>

select distinct ?ttftime ?ttfstoch ?ttrtime ?ttrstoch
WHERE { 
	# The selected element is a user-defined parameter
	BIND(URI("#failuremode#") AS ?failstate) . 	
	?intr fsm:Target ?failstate .
	?intr fsm:TransitionGuard/fsm:GuardEvent/ifcext:hasTimeToHappen ?ttf .
	?failstate osph:do/ifc:taskTime_IfcTask ?ttr .
	optional{
		?ttf osph:isQuantitySampledFrom ?ttfstoch.
	}
	optional{
		?ttf express:hasDouble ?ttftime .
	}
	optional{
		?ttr ifcext:hasStochasticDuration/osph:isQuantitySampledFrom ?ttrstoch.
	}
	optional{
		?ttr ifc:scheduleDuration_IfcTaskTime/express:hasString ?ttrtime .
	}
}
