DELETE{
	GRAPH ?g1{ ?elemhist ?b1 ?c1 .} 
	GRAPH ?g2{ ?a2 ?b2 ?elemhist .} 
	
}
WHERE { 

	#?elemhistclass rdfs:subClassOf* osph:ObjectHistory .
	#?elemhist rdf:type ?elemhistclass . 
	
	# Get all histories
	?hist (osph:isDecomposedByHistory|^osph:decomposesHistory)* ?elemhist .
	
	{ GRAPH ?g1{ ?elemhist ?b1 ?c1 .} }
	UNION { GRAPH ?g2{ ?a2 ?b2 ?elemhist .} }	
}
