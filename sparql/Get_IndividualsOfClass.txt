SELECT distinct ?individ ?itsClass WHERE { 
  ?individ rdf:type/rdfs:subClassOf* #class# . 
  ?individ rdf:type owl:NamedIndividual .
  ?individ rdf:type ?itsClass .
  FILTER ( ?itsClass != owl:NamedIndividual ) .
}
